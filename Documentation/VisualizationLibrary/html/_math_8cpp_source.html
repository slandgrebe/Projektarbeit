59 bool Math::segmentIntersectsTriangle(glm::vec3 start, glm::vec3 end,
60  glm::vec3 vertexA, glm::vec3 vertexB, glm::vec3 vertexC) {
61 
62  // Dreieckseiten und -normale
63  glm::vec3 edgeAB = vertexB - vertexA; // Seite des Dreiecks von Punkt A bis Punkt B
64  glm::vec3 edgeAC = vertexC - vertexA; // Seite des Dreiecks von Punkt A bis Punkt C
65  glm::vec3 normal = glm::cross(edgeAB, edgeAC); // Normale des Dreiecks
66 
67  if (normal == glm::vec3(0)) { // triangle is degenerate
68  /*Log().warning() << "degenerate triangle: a[" << vertexA.x << "/" << vertexA.y << "/" << vertexA.z << "] " <<
69  "b[" << vertexB.x << "/" << vertexB.y << "/" << vertexB.z << "] " <<
70  "c[" << vertexC.x << "/" << vertexC.y << "/" << vertexC.z << "]";*/
71  return false; // do not deal with this case
72  }
73 
74  glm::vec3 direction = end - start; // Richtung der Strecke
75  glm::vec3 w0 = start - vertexA; // ray vector: Vektor von Punkt A des Dreiecks bis Anfang der Strecke
76  float a = -glm::dot(normal, w0); // -Cosinus des Winkels zwischen Normale und w0
77  float b = glm::dot(normal, direction); // Cosinus des Winkels zwischen Dreieck-Normale und Strecken-Richtung
78  if (b < NEAR_ZERO && b > -NEAR_ZERO) { // Strecke ist parallel zur Dreieck Ebene
79  if (a == 0) { // Strecke liegt in der Dreieck Ebene
80  if (segmentIntersection(start, end, vertexA, vertexB)) { // schneidet strecke die Seite AB
81  return true;
82  }
83  else if (segmentIntersection(start, end, vertexA, vertexC)) { // schneidet strecke die Seite AC
84  return true;
85  }
86  else if (segmentIntersection(start, end, vertexB, vertexC)) { // schneidet strecke die Seite BC
87  return true;
88  }
89  return false; // keine berührungspunkte
90  }
91  else {
92  return false; // Strecke berührt die Dreieck Ebene nicht
93  }
94  }
95 
96  // Schnittpunkt der Strecke mit der Dreieckebene
97  float r = a / b; // params to calc ray-plane intersect
98  if (r < 0.0 // ray goes away from triangle
99  && r > 1.0) { // for a segment, also test if (r > 1.0) => no intersect
100 
101  return false; // => no intersect
102  }
103 
104  glm::vec3 intersection = start + r * direction; // intersect point of ray and plane
105 
106  // is I inside T?
107  float uu, uv, vv, wu, wv, D;
108  uu = glm::dot(edgeAB, edgeAB);
109  uv = glm::dot(edgeAB, edgeAC);
110  vv = glm::dot(edgeAC, edgeAC);
111  glm::vec3 w = intersection - vertexA; // ray vector
112  wu = glm::dot(w, edgeAB);
113  wv = glm::dot(w, edgeAC);
114  D = uv * uv - uu * vv;
115 
116  // get and test parametric coords
117  float s, t;
118  s = (uv * wv - vv * wu) / D;
119  if (s < 0.0 || s > 1.0) { // I is outside T
120  return false;
121  }
122  t = (uv * wu - uu * wv) / D;
123  if (t < 0.0 || (s + t) > 1.0) { // I is outside T
124  return false;
125  }
126 
127  return true; // I is in T
128 }
