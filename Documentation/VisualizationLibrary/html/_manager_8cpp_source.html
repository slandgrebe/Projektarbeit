426 void Manager::doCollisionDetection(void) {
427  //{ 1:{2, 3}, 2 : {1, 2} }
428  //1:2,3;2:1,2
429  std::stringstream collisions;
430  const char* separator1 = "";
431  const char* separator2 = "";
432  glm::vec3 cameraPosition = graphics::GraphicEngine::getInstance()->camera()->position();
433 
434  // Modelle
435  std::map<GLuint, model::AssimpModel*>::iterator it;
436  for (it = assimpModelList.begin(); it != assimpModelList.end(); it++) {
437  model::AssimpModel* modelA = (*it).second;
438  collisions << separator1 << it->first << ":";
439  separator1 = ";";
440 
441  glm::vec3 a = modelA->position();
442  if (modelA->attachedToCamera()) {
443  a += cameraPosition;
444  }
445  float ar = modelA->boundingSphereRadius();
446 
447  std::map<GLuint, model::AssimpModel*>::iterator it2;
448  for (it2 = assimpModelList.begin(); it2 != assimpModelList.end(); it2++) {
449  model::AssimpModel* modelB = (*it2).second;
450 
451  if (modelA == modelB) {
452  continue;
453  }
454 
455 
456  glm::vec3 b = modelB->position();
457  if (modelB->attachedToCamera()) {
458  b += cameraPosition;
459  }
460 
461  float br = modelB->boundingSphereRadius();
462 
463  if (pow(a.x - b.x, 2) + pow(a.y - b.y, 2) + pow(a.z - b.z, 2) < pow(ar + br, 2)) {
464  collisions << separator2 << it2->first;
465  separator2 = ",";
466  }
467  }
468 
469  separator2 = "";
470  }
471 
472  this->m_collisions = collisions.str();
473 }
