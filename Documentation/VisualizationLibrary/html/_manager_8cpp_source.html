506 void Manager::doCollisionDetection(void) {
507  //{ 1:{2, 3}, 2 : {1, 2} }
508  //1:2,3;2:1,2
509  std::stringstream collisions;
510  const char* separator1 = "";
511  const char* separator2 = "";
512  glm::vec3 cameraPosition = graphics::GraphicEngine::getInstance()->camera()->position();
513 
514  // Modelle
515  std::map<GLuint, model::AssimpModel*>::iterator it;
516  for (it = assimpModelList.begin(); it != assimpModelList.end(); it++) {
517  model::AssimpModel* modelA = (*it).second;
518  collisions << separator1 << it->first << ":";
519  separator1 = ";";
520 
521  glm::vec3 a = modelA->position();
522  if (modelA->attachedToCamera()) {
523  a += cameraPosition;
524  }
525  float ar = modelA->boundingSphereRadius();
526 
527  std::map<GLuint, model::AssimpModel*>::iterator it2;
528  for (it2 = assimpModelList.begin(); it2 != assimpModelList.end(); it2++) {
529  model::AssimpModel* modelB = (*it2).second;
530 
531  if (modelA == modelB) {
532  continue;
533  }
534 
535 
536  glm::vec3 b = modelB->position();
537  if (modelB->attachedToCamera()) {
538  b += cameraPosition;
539  }
540 
541  float br = modelB->boundingSphereRadius();
542 
543  if (pow(a.x - b.x, 2) + pow(a.y - b.y, 2) + pow(a.z - b.z, 2) < pow(ar + br, 2)) {
544  collisions << separator2 << it2->first;
545  separator2 = ",";
546  }
547  }
548 
549  separator2 = "";
550  }
551 
552  this->m_collisions = collisions.str();
553 }
