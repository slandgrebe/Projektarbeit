186 bool AssimpModel::initSingleMesh(const int meshIndex, const aiMesh* mesh) {
187  Log().debug() << " Mesh #" << meshIndex << "; Anzahl Seiten: " << mesh->mNumFaces ;
188 
189  meshList[meshIndex].materialIndex = mesh->mMaterialIndex;
190 
191  std::vector<Vertex> vertices;
192  std::vector<unsigned int> indices;
193 
194  const aiVector3D zero3D(0.0f, 0.0f, 0.0f);
195 
196  // Vertices
197  for (unsigned int i = 0; i < mesh->mNumVertices; i++) {
198  const aiVector3D* pos = &(mesh->mVertices[i]);
199  const aiVector3D* normal = &(mesh->mNormals[i]);
200  const aiVector3D* texCoord = mesh->HasTextureCoords(0) ? &(mesh->mTextureCoords[0][i]) : &zero3D;
201 
202  Vertex v( glm::vec3(pos->x, pos->y, pos->z),
203  glm::vec2(texCoord->x, texCoord->y),
204  glm::vec3(normal->x, normal->y, normal->z));
205 
206  // bounding Sphere
207  float radius = sqrt(pos->x * pos->x + pos->y * pos->y + pos->z * pos->z); // pythagoras
208  if (radius > m_boundingSphereRadius) m_boundingSphereRadius = radius;
209  //if (std::abs(pos->x) > m_boundingSphereRadius) m_boundingSphereRadius = std::abs(pos->x);
210  //if (std::abs(pos->y) > m_boundingSphereRadius) m_boundingSphereRadius = std::abs(pos->y);
211  //if (std::abs(pos->z) > m_boundingSphereRadius) m_boundingSphereRadius = std::abs(pos->z);
212 
213  vertices.push_back(v);
214  }
215 
216  // Faces
217  for (unsigned int i = 0; i < mesh->mNumFaces; i++) {
218  const aiFace& face = mesh->mFaces[i];
219  assert(face.mNumIndices == 3);
220  indices.push_back(face.mIndices[0]);
221  indices.push_back(face.mIndices[1]);
222  indices.push_back(face.mIndices[2]);
223  }
224 
225  meshList[meshIndex].init(vertices, indices);
226 
227  // collision detection
228  for (unsigned int i = 0; i < indices.size(); i++) { // alle punkte des models durchgehen
229  glm::vec3 point = vertices[indices[i]].position;
230 
231  if (point.x > xMax) xMax = point.x;
232  if (point.x < xMin) xMin = point.x;
233  if (point.y > yMax) yMax = point.y;
234  if (point.y < yMin) yMin = point.y;
235  if (point.z > zMax) zMax = point.z;
236  if (point.z < zMin) zMin = point.z;
237  }
238 
239  return true;
240 }
