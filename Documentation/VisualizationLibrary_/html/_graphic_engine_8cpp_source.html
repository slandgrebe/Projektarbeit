169 void GraphicEngine::processQueue() {
170  while (modelQueue.hasMore()) {
171  modelQueueEntry e = modelQueue.dequeue();
172  model::AssimpModel* model = new model::AssimpModel;
173  if (model->loadModel(e.filename)) {
174  Manager::getInstance()->addToModelList(e.modelId, model);
175  }
176  else {
177  Log().error() << "Could not create model. id[" << e.modelId << "] filename[" << e.filename << "]" ;
178  }
179  }
180 
181  while (squareQueue.hasMore()) {
182  modelQueueEntry e = squareQueue.dequeue();
183  model::Square* model = new model::Square;
184  if (model->loadFromFile(e.filename)) {
185  Manager::getInstance()->addToSquareList(e.modelId, model);
186  }
187  else {
188  Log().error() << "Could not create square. id[" << e.modelId << "] filename[" << e.filename << "]";
189  }
190  }
191 
192  while (textQueue.hasMore()) {
193  modelQueueEntry e = textQueue.dequeue();
194  gui::Text* text = new gui::Text;
195  if (text->init(e.filename)) {
196  Manager::getInstance()->addToTextList(e.modelId, text);
197  }
198  else {
199  Log().error() << "Could not create text. id[" << e.modelId << "] text[" << e.filename << "]";
200  }
201  }
202 
203  while (buttonQueue.hasMore()) {
204  modelQueueEntry e = buttonQueue.dequeue();
205  gui::Button* button = new gui::Button;
206  if (button->init(e.filename)) {
207  Manager::getInstance()->addToButtonList(e.modelId, button);
208  }
209  else {
210  Log().error() << "Could not create button. id[" << e.modelId << "] fontname[" << e.filename << "]";
211  }
212  }
213 
214  while (disposeQueue.hasMore()) {
215  GLuint modelId = disposeQueue.dequeue();
216  Manager::getInstance()->remove(modelId);
217  }
