306  private List<LevelSegment> ChooseRandomSegments(JumpAndRun.Difficulty difficulty, int lengthInSeconds, double speed)
307  {
308 
309  difficulty = JumpAndRun.Difficulty.Difficult;
310  double currentLength = 0; // in meter
311 
312  int minDifficulty = AllAvailableSegments.Min(e => e.Severity); // einfachste vorhandene Schwierigkeit
313  int maxDifficulty = AllAvailableSegments.Max(e => e.Severity); // schwerste vorhandene Schwierigkeit
314 
315  double nDifficulty = 0; // schwierigkeit auf einer Skala von min - max
316  switch (difficulty)
317  {
318  case JumpAndRun.Difficulty.Easy: nDifficulty = Math.Round((double)(maxDifficulty - minDifficulty) / 4.0) + minDifficulty; break; // 25%
319  case JumpAndRun.Difficulty.Normal: nDifficulty = Math.Round((double)(maxDifficulty - minDifficulty) / 2.0) + minDifficulty; break; // 50%
320  case JumpAndRun.Difficulty.Difficult: nDifficulty = Math.Round((double)(maxDifficulty - minDifficulty) / 4.0 * 3) + minDifficulty; break; // 75%
321  default: nDifficulty = Math.Round((double)(maxDifficulty - minDifficulty) / 2) + minDifficulty; break;
322  }
323 
324  double standardDeviation = (maxDifficulty - minDifficulty) * 0.3; // Standardabweichung
325 
326  // bei nur einem segment...
327  if (standardDeviation <= 0)
328  {
329  standardDeviation = 1;
330  }
331 
332  List<LevelSegment> segmentList = new List<LevelSegment>();
333  while(true)
334  {
335  // Zufallszahl nach Gaussscher Normalverteilung mit nDifficulty als Durchschnitt
336  double d = (int)Math.Round(RandomNumberGenerator.GetNormal(nDifficulty, standardDeviation));
337 
338  // min und max kontrollieren

348  // so lange Segmente hinzuf√ºgen, bis die gew√ºnschte L√§nge erreicht wird.
349  if ((currentLength + segment.Length) / speed < lengthInSeconds)
350  {
351  currentLength += segment.Length;

356  }
357  }

406  private LevelSegment GetSegmentWithDifficulty(double difficulty)
407  {
408  double smallestDifference = 100000;
409  List<LevelSegment> list = new List<LevelSegment>();
410 
411  foreach (LevelSegment segment in AllAvailableSegments)

414  if (currentDifference < smallestDifference)
415  {
416  smallestDifference = currentDifference;
417  list.Clear();
418  list.Add(segment);
419  }
420  else if (currentDifference == smallestDifference)
421  {
422  list.Add(segment);
423  }
424  }
425
