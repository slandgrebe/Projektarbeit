53  // Produce a uniform random sample from the open interval (0, 1).
54  // The method will not return either end point.
55  public static double GetUniform()
56  {
57  // 0 <= u < 2^32
58  uint u = GetUint();
59  // The magic number below is 1/(2^32 + 2).
60  // The result is strictly between 0 and 1.
61  return (u + 1.0) * 2.328306435454494e-10;
62  }

64  // This is the heart of the generator.
65  // It uses George Marsaglia's MWC algorithm to produce an unsigned integer.
66  // See http://www.bobwheeler.com/statistics/Password/MarsagliaPost.txt
67  private static uint GetUint()
68  {
69  m_z = 36969 * (m_z & 65535) + (m_z >> 16);
70  m_w = 18000 * (m_w & 65535) + (m_w >> 16);
71  return (m_z << 16) + m_w;
72  }

113  public static double GetGamma(double shape, double scale)
114  {
115  // Implementation based on "A Simple Method for Generating Gamma Variables"
116  // by George Marsaglia and Wai Wan Tsang. ACM Transactions on Mathematical Software
117  // Vol 26, No 3, September 2000, pages 363-372.
118 
119  double d, c, x, xsquared, v, u;
120 
121  if (shape >= 1.0)
122  {
123  d = shape - 1.0 / 3.0;
124  c = 1.0 / Math.Sqrt(9.0 * d);
125  for (; ; )
126  {
127  do
128  {
129  x = GetNormal();
130  v = 1.0 + c * x;
131  }
132  while (v <= 0.0);
133  v = v * v * v;
134  u = GetUniform();
135  xsquared = x * x;
136  if (u < 1.0 - .0331 * xsquared * xsquared || Math.Log(u) < 0.5 * xsquared + d * (1.0 - v + Math.Log(v)))
137  return scale * d * v;
138  }
139  }
140  else if (shape <= 0.0)
141  {
142  string msg = string.Format("Shape must be positive. Received {0}.", shape);
143  throw new ArgumentOutOfRangeException(msg);
144  }
145  else
146  {
147  double g = GetGamma(shape + 1.0, 1.0);
148  double w = GetUniform();
149  return scale * g * Math.Pow(w, 1.0 / shape);
150  }
151  }
